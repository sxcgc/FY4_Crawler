        $(function () {
            // 保存最近浏览的20个产品
            var cook = "";
            if ($.cookie("ViewHistory") != undefined && $.cookie("ViewHistory") != null && $.cookie("ViewHistory") != "") {
                // 如果已经存在该产品
                var history = $.cookie("ViewHistory").split(',');
                var newProduct = $("#hidProductID").val().split(',');
                var coo = $.cookie("ViewHistory");
                $.each(history, function (i, historyItem) {
                    $.each(newProduct, function (j, newItem) {
                        if (historyItem == newItem)
                            coo = coo.replace((historyItem + ","), "");//去掉重复
                    })
                });
                cook = $("#hidProductID").val() + "," + coo;
            }
            else {
                cook = $("#hidProductID").val() + ",";
            }
            // 超过5个产品找最新的5个
            var productid = "";
            if (cook.split(',').length > 5) {
                $.each(cook.split(','), function (i, item) {
                    if (i < 5 && item != "") {
                        productid += item;
                        productid += ",";
                    }
                });
            }
            else
                productid = cook;
            $.cookie("ViewHistory", productid, { path: '/', expires: 1 });
        });

        // 获取没有选中的条件
        function getNoWhere() {
            var idList = "", id = "";
            $.each($("#flSearch .data div"), function () {
                id = "";
                $.each($("#" + $(this).attr("id") + " input:checkbox"), function (i, item) {
                    if (!$(item).is(":checked"))
                        id += $(item).attr("id").split('+')[1] + ",";
                });
                if (id != "") {
                    id = id.substr(0, id.length - 1);
                    idList += (id + "@" + $(this).attr("data-value") + "|");
                }
            });
            return idList;
        }

        // path:url地址，params：参数，callback执行成功的回调函数，isJson 是否转化为json对象
        function ajax(path, params, callback, isJson, id, ckbValue) {
            $.ajax({
                type: "POST",
                contentType: "application/json;charset=utf-8",
                url: path,
                data: params,
                dataType: 'json',
                success: function (data) {
                    var strHtml = "";
                    var json = "";
                    if (data.d == "nologin") {
                        strHtml += "获取数据有误";
                    }
                    else if (data.d == null) {
                        strHtml += "无";
                    }
                    else {
                        if (isJson) {
                            if (data.d != "") {
                                json = eval("(" + data.d + ")");
                            }
                        }
                        else {
                            json = data.d;
                        }
                    }
                    callback(json, strHtml, id, ckbValue);
                },
                error: function (XMLHttpRequest, textStatus, errorThrown) {
                    alert('error：' + XMLHttpRequest.responseText + "|" + params);
                }
            });
    }

        var $curObj = $(objSelector);

                $img.attr("title", "Little");
                $curList.show();
                restoreContSpanText($curList.eq(showCount - 1));
                changeContSpanText($curList.eq($curList.length - 1));
                $curList.eq($curList.length - 1).append($img);

                $img.attr("title", "More");
                $curList.hide();
                $curList.slice(0, showCount).show();
                restoreContSpanText($curList.eq($curList.length - 1));
                changeContSpanText($curList.eq(showCount - 1));
                $curList.eq(showCount - 1).append($img);
            });
            changeContSpanText($curList.eq(showCount - 1));
            $curList.eq(showCount - 1).append($img);
        }
    }

    function getCookie(name) {
        var bikky = document.cookie;
        name += "=";
        var i = 0;
        while (i < bikky.length) {
            var offset = i + name.length;
            if (bikky.substring(i, offset) == name) {
                var endstr = bikky.indexOf(";", offset);
                if (endstr == -1) endstr = bikky.length;
                return unescape(bikky.substring(offset, endstr));
            }
            i = bikky.indexOf(" ", i) + 1;
            if (i == 0) break;
        }
        return null;
    }

    function ajax1(path, params, callback, isJson, controlid) {
        $.ajax({
            type: "POST",
            contentType: "application/json;charset=utf-8",
            url: path,
            data: params,
            dataType: 'json',
            success: function (data) {
                var strHtml = "";
                if (data.d == "error") {
                    strHtml += "获取数据有误";
                }
                else if (data.d == "") {
                    strHtml += "无";
                }
                else {
                    var json = "";
                    if (isJson)
                        json = eval("(" + data.d + ")");
                    else
                        json = data.d;
                    callback(json, strHtml, controlid);

                }
            },
            error: function (XMLHttpRequest, textStatus, errorThrown) {
                alert('error：' + XMLHttpRequest.responseText);
            }
        });
}
